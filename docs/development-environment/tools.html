Add Zappr Config

Add a .zappr.yaml (Documentation) in the top-level of the repository:

autobranch:
  pattern: '{number}-{title}'
  length: 60
commit:
  message:
    patterns:
      - '#[0-9]+'
approvals:
  minimum: 1
  ignore: none
  pattern: "^(:\\+1:|üëç)$"
  veto:
    pattern: "^(:\\-1:|üëé)$"
specification:
  title:
    minimum-length:
      enabled: true
      length: 8
  body:
    minimum-length:
      enabled: true
      length: 8
    contains-url: true
    contains-issue-number: true
  template:
    differs-from-body: true
pull-request:
  labels:
    additional: true
